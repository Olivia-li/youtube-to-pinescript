Here's a Pine script for the trading strategy you described:

```
//@version=4
strategy("Ultimate Trading Strategy", overlay=true)

// Signs of the Time Indicator
signsOfTheTime = input(title="Signs of the Time Indicator", type=input.bool, defval=true)
signsOfTheTimeColor = color.green
signsOfTheTimeColor := signsOfTheTime ? iff(close > open, color.green, color.red) : color.gray
plot(signsOfTheTimeColor, title="Signs of the Time Indicator", style=plot.style_area, transp=50)

// Twin Range Filter Indicator
twinRangeFilter = input(title="Twin Range Filter Indicator", type=input.bool, defval=true)
trfHigh = highest(high, 14)
trfLow = lowest(low, 14)
trfRange = trfHigh - trfLow
trfThreshold = trfRange * 0.75
trfSignal = trfRange > trfThreshold ? 1 : 0
plotshape(twinRangeFilter ? trfSignal : na, title="Twin Range Filter Indicator", style=shape.triangleup, location=location.belowbar, color=color.green, size=size.small)

// Exponential Moving Average
ema = input(title="Exponential Moving Average", type=input.bool, defval=true)
emaValue = ema(close, 14)
plot(emaValue, title="Exponential Moving Average", color=color.blue)

// Buy Signal
buySignal = signsOfTheTimeColor == color.green and trfSignal == 1 and close > emaValue
plotshape(buySignal ? close : na, title="Buy Signal", style=shape.circle, location=location.belowbar, color=color.green, size=size.small)

// Sell Signal
sellSignal = signsOfTheTimeColor == color.red and trfSignal == 1 and close < emaValue
plotshape(sellSignal ? close : na, title="Sell Signal", style=shape.circle, location=location.abovebar, color=color.red, size=size.small)

// Strategy
if buySignal
    strategy.entry("Buy", strategy.long)
    strategy.exit("Stop Loss/Take Profit", "Buy", stop=close * 0.985, limit=close * 1.015)
if sellSignal
    strategy.entry("Sell", strategy.short)
    strategy.exit("Stop Loss/Take Profit", "Sell", stop=close * 1.015, limit=close * 0.985)
``` 

This Pine script implements the three indicators you mentioned: Signs of the Time Indicator, Twin Range Filter Indicator, and Exponential Moving Average. It also includes buy and sell signals based on those indicators and sets a stop loss and take profit for each trade.
