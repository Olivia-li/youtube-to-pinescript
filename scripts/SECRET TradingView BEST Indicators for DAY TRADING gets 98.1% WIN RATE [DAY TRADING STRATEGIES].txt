```
//@version=4
strategy("Ultimate Trading Strategy", overlay=true)

// Signs of the Time Indicator
signs_of_time = input(title="Signs of the Time Indicator", type=input.bool, defval=true)
signs_of_time_color = signs_of_time ? ema(close, 8) > ema(close, 21) ? color.green : color.red : na
plot(signs_of_time ? ema(close, 8) : na, color=signs_of_time_color, linewidth=2, title="Signs of the Time")

// Twin Range Filter Indicator
twin_range_filter = input(title="Twin Range Filter Indicator", type=input.bool, defval=true)
trf_high = highest(high, 14)
trf_low = lowest(low, 14)
trf_range = trf_high - trf_low
trf_condition = (trf_range < trf_range[1]) ? trf_range : trf_range[1]
trf_value = trf_condition * 0.8
plot(twin_range_filter ? trf_value : na, color=color.blue, linewidth=2, title="Twin Range Filter")

// Exponential Moving Average
ema_indicator = input(title="Exponential Moving Average", type=input.bool, defval=true)
ema_value = ema(close, 50)
plot(ema_indicator ? ema_value : na, color=color.orange, linewidth=2, title="Exponential Moving Average")

// Buy and Sell Signals
buy_signal = signs_of_time_color == color.green and close > ema_value and close > trf_value
sell_signal = signs_of_time_color == color.red and close < ema_value and close < trf_value

// Strategy Execution
if buy_signal
    strategy.entry("Buy", strategy.long)
    strategy.exit("Stop Loss/Take Profit", "Buy", stop=close * 0.985, limit=close * 1.015)
if sell_signal
    strategy.entry("Sell", strategy.short)
    strategy.exit("Stop Loss/Take Profit", "Sell", stop=close * 1.015, limit=close * 0.985)
```
