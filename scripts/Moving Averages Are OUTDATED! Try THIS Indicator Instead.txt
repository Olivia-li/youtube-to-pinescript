//@version=4
study(title="Zero Lag EMA Strategy", shorttitle="ZEMA")

// Define inputs
len = input(title="Length", type=input.integer, defval=20)
mult = input(title="Multiplier", type=input.float, defval=2.5)
stop_mult = input(title="Stop Loss Multiplier", type=input.float, defval=1.5)

// Define ZEMA function
zema(src, len) =>
    alpha = 2 / (len + 1)
    ema1 = ema(src, len)
    ema2 = ema(ema1, len)
    zema = (alpha * 2 * ema1 - (1 - alpha) * ema2) ^ 2

// Calculate ZEMA lines
zema_blue = zema(close, len)
zema_red = zema(zema_blue, len)
zema_green = zema(zema_red, len)

// Plot ZEMA lines
plot(zema_blue, color=color.blue, linewidth=2)
plot(zema_red, color=color.red, linewidth=2)
plot(zema_green, color=color.green, linewidth=2)

// Define long entry conditions
long_entry = crossover(zema_red, zema_blue) and crossover(zema_green, zema_red) and close > zema_red

// Define short entry conditions
short_entry = crossunder(zema_red, zema_blue) and crossunder(zema_red, zema_green) and close < zema_red

// Plot entry signals
plotshape(long_entry, style=shape.triangleup, location=location.belowbar, color=color.green, size=size.small)
plotshape(short_entry, style=shape.triangledown, location=location.abovebar, color=color.red, size=size.small)

// Define stop loss and take profit levels
stop_loss = low - atr(len) * stop_mult
take_profit = close + (close - stop_loss) * mult

// Plot stop loss and take profit levels
plot(stop_loss, color=color.red, title="Stop Loss")
plot(take_profit, color=color.green, title="Take Profit")

// Execute long trade
if long_entry
    strategy.entry("Long", strategy.long)
    strategy.exit("Long Stop Loss", "Long", stop=stop_loss, when=strategy.position_size < 0)
    strategy.exit("Long Take Profit", "Long", limit=take_profit, when=strategy.position_size > 0)

// Execute short trade
if short_entry
    strategy.entry("Short", strategy.short)
    strategy.exit("Short Stop Loss", "Short", stop=stop_loss, when=strategy.position_size > 0)
    strategy.exit("Short Take Profit", "Short", limit=take_profit, when=strategy.position_size < 0)
